var progress=0;
function startDeal(){
    if(progress>100){
        $(".activeBar").html("完成");
            return;
        }
    if(progress<=100){
        setTimeout("startDeal()",500);			
        $(".activeBar").css("width",progress+"%").html(progress+"%");
        progress+=5;
    }
}

bootstrap.css，ui.css

说明

进度条须引用.progress整体样式，.progress-bar-success和.progress-bar-danger分别设置成功进度条和危险进度条的样式。 危险进度条上的纹理效果须引用. progress-bar-striped样式。点击button可以查看进度条的动态效果。

aria-valuenow设置现在状态，aria-valuemin设置状态最小值，aria-valuemax设置状态达到的最大值。
